userId: Globally unique, assigned by Firebase
groupId: Globally unique, auto generated 
groupName: Globally unique, chosen by the owner
entityId: Unique within a group across channelId, eventId and articleId
privilege: access right (5:owner, 4:admin, 3:mentor, 2:subscriber, 1:member, 0:guest)

/users/{userId}
    created: date
    lastAccessed: date

/groupnames/{groupName}
    groupId: string

/groups/{groupId}
    created: date
    title: string
    description: string
    groupName: string
    logo: string
    banner: string
    privileges: {
        membership: {
            closed: boolean
        }
        invitation: {
            privilege: number
        }
        channel: {
            privilege: number
        }
        event: {
            privilege: number
        }
        article: {
            privilege: number
        }
    },
    /members/{userId}
        created: date (duplicated)
        privilege: number (duplicated)
        name: string
        lastAccessed: date
        desciption: string
        email: string

    /privileges/{userId}
        value: number (5:owner, 4:admin, 3:teacher, 2:subscriber, 1:member)
        created: date

    /invitations/{invitationId}
        expires: date
        once: boolean

    /entities/{entityId}  # channelId, eventId, articleId
        onwer: userId

    /channels/{channelId}
        created: date
        title: string
        owner: userId
        privilege: number
        /messages/{messageId}
            created: date
            message: string
            users: {
                userId: userId
                name: string ###
            }
            editted: boolean

    /events/{eventId}
        title: string
        description: string
        date: date
        onwer: userId
        privilege: number
        capacity: number
        /attendees/{userId}
            created: date
            name: string ###
            accepted: boolean
            attended: boolean

    /articles/{articleId}
        title: string
        owner: userId
        type: string # "news", "blog", "announcement", "page"
        privilege: number
        sections: [sectionId, ...]
        /authors/{userId}
            name: string ###
        /sections/{sectionId}
            text: string
            image: string
            video: string
            type: string
            updated: date
